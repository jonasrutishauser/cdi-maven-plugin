54  /**
55   * Validates the CDI configuration of a war or ear.
56   * 
57   * @author Jonas Rutishauser
58   */
59  @Mojo(name = "validate", defaultPhase = LifecyclePhase.VERIFY, requiresProject = true, threadSafe = true)
60  public class ValidateMojo extends AbstractMojo {
61  
62      private static final String SMALLRYE_GROUP_ID = "io.smallrye";
63  
64      private final MavenSession session;
65  
66      private final MavenProject project;
67  
68      private final ArchiverManager archiverManager;
69  
70      private final DependencyResolver dependencyResolver;
71  
72      private final Map<String, String> oldSystemProperties = new HashMap<>();
73  
74      @Parameter(defaultValue = "${project.build.directory}/cdi", readonly = true)
75      private File workDirectory;
76  
77      @Parameter
78      private Map<String, String> systemProperties = new HashMap<>();
79  
80      @Parameter
81      private boolean microprofile = false;
82  
83      @Inject
84      public ValidateMojo(MavenSession session, MavenProject project, ArchiverManager archiverManager,
85              DependencyResolver dependencyResolver) {
86          this.session = session;
87          this.project = project;
88          this.archiverManager = archiverManager;
89          this.dependencyResolver = dependencyResolver;
90      }
91  
92      @Override
93      public void execute() throws MojoExecutionException, MojoFailureException {
94          fixCdiApiInClassloader();
95          ArchiveUtil util;
96          if ("ear".equals(project.getPackaging())) {
97              util = new EarUtil(archiverManager);
98          } else if ("war".equals(project.getPackaging())) {
99              util = new WarUtil(archiverManager);
100         } else {
101             return;
102         }
103         try {
104             util.init(workDirectory, project.getArtifact().getFile(), getParentClassLoader());
105         } catch (MalformedURLException e) {
106             throw new MojoExecutionException("failed to load ear", e);
107         }
108         systemProperties.forEach((key, value) -> oldSystemProperties.put(key, System.setProperty(key, value)));
109         oldSystemProperties.put("org.jboss.logging.provider",
110                 System.setProperty("org.jboss.logging.provider", "slf4j"));
111         CDI11Bootstrap bootstrap = new WeldBootstrap();
112         ClassLoader oldContextClassLoader = Thread.currentThread().getContextClassLoader();
113         try {
114             Thread.currentThread().setContextClassLoader(util.getClassLoader());
115             bootstrap.startContainer(UUID.randomUUID().toString(), Environments.EE, util.createDeployment(bootstrap));
116             bootstrap.startInitialization();
117             bootstrap.deployBeans();
118             bootstrap.validateBeans();
119             bootstrap.endInitialization();
120         } catch (Exception e) {
121             throw new MojoFailureException("CDI error found: " + e.getMessage(), e);
122         } finally {
123             Thread.currentThread().setContextClassLoader(oldContextClassLoader);
124             bootstrap.shutdown();
125             oldSystemProperties.forEach((key, value) -> {
126                 if (value == null) {
127                     System.clearProperty(key);
128                 } else {
129                     System.setProperty(key, value);
130                 }
131             });
132         }
133     }
134 
135     private void fixCdiApiInClassloader() {
136         if (getClass().getClassLoader() instanceof ClassRealm) {
137             ClassRealm ownRealm = (ClassRealm) getClass().getClassLoader();
138             if (ownRealm.getImportClassLoader("javax.enterprise.inject") instanceof ClassRealm) {
139                 ownRealm.importFrom(new EmptyClassLoader(), "javax.enterprise.inject");
140             }
141         }
142     }
143 
144     private ClassLoader getParentClassLoader() throws MojoExecutionException {
145         if (microprofile) {
146             List<URL> urls = new ArrayList<>();
147             try {
148                 List<Dependency> managedDependencies = project.getDependencyManagement() == null ? new ArrayList<>() : project.getDependencyManagement().getDependencies();
149                 for (ArtifactResult resolved : dependencyResolver.resolveDependencies(
150                         session.getProjectBuildingRequest(), getMicroprofileImplementations(),
151                         managedDependencies, null)) {
152                     urls.add(resolved.getArtifact().getFile().toURI().toURL());
153                 }
154             } catch (DependencyResolverException | MalformedURLException e) {
155                 throw new MojoExecutionException("Failed to retrieve microprofile dependencies", e);
156             }
157             return new URLClassLoader(urls.toArray(new URL[urls.size()]), getClass().getClassLoader());
158         }
159         return getClass().getClassLoader();
160     }
161 
162     private Collection<Dependency> getMicroprofileImplementations() {
163         Collection<Dependency> implementations = new ArrayList<>();
164         Dependency configImpl = new Dependency();
165         configImpl.setGroupId(SMALLRYE_GROUP_ID);
166         configImpl.setArtifactId("smallrye-config");
167         configImpl.setVersion("1.5.0");
168         implementations.add(configImpl);
169         Dependency faultToleranceApi = new Dependency();
170         faultToleranceApi.setGroupId("org.eclipse.microprofile.fault-tolerance");
171         faultToleranceApi.setArtifactId("microprofile-fault-tolerance-api");
172         faultToleranceApi.setVersion("2.1.1");
173         implementations.add(faultToleranceApi);
174         Dependency healthApi = new Dependency();
175         healthApi.setGroupId("org.eclipse.microprofile.health");
176         healthApi.setArtifactId("microprofile-health-api");
177         healthApi.setVersion("2.2");
178         implementations.add(healthApi);
179         Dependency metricsApi = new Dependency();
180         metricsApi.setGroupId("org.eclipse.microprofile.metrics");
181         metricsApi.setArtifactId("microprofile-metrics-api");
182         metricsApi.setVersion("2.3");
183         implementations.add(metricsApi);
184         Dependency metricsImpl = new Dependency();
185         metricsImpl.setGroupId(SMALLRYE_GROUP_ID);
186         metricsImpl.setArtifactId("smallrye-metrics");
187         metricsImpl.setVersion("2.4.2");
188         implementations.add(metricsImpl);
189         Dependency jwtImpl = new Dependency();
190         jwtImpl.setGroupId(SMALLRYE_GROUP_ID);
191         jwtImpl.setArtifactId("smallrye-jwt");
192         jwtImpl.setVersion("2.2.0");
193         implementations.add(jwtImpl);
194         Dependency restClientImpl = new Dependency();
195         restClientImpl.setGroupId(SMALLRYE_GROUP_ID);
196         restClientImpl.setArtifactId("smallrye-rest-client");
197         restClientImpl.setVersion("1.2.2");
198         implementations.add(restClientImpl);
199         return implementations;
200     }
