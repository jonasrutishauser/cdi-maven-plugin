38  @RunWith(MavenJUnitTestRunner.class)

41  
42      @Rule
43      public final TestResources resources = new TestResources();
44  
45      public final MavenRuntime mavenRuntime;
46  
47      public ValidateMojoIT(MavenRuntimeBuilder builder) throws Exception {
48          this.mavenRuntime = builder.withCliOptions("-B", "-X").build();
49      }
50  
51      @Test
52      public void verifySimpleValid() throws Exception {
53          File basedir = resources.getBasedir("simple");
54  
55          MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "verify");
56  
57          result.assertErrorFreeLog().assertLogText("Session bean [class test.TestBean");
58      }
59  
60      @Test
61      public void verifySimpleNotValid() throws Exception {
62          File basedir = resources.getBasedir("simple");
63          Files.delete(basedir.toPath().resolve("simple-java/src/main/resources/META-INF/beans.xml"));
64  
65          MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "verify");
66  
67          result.assertLogText("Unsatisfied dependencies for type Foo with qualifiers @Default")
68                  .assertLogText("private test.TestBean.foo");
69      }

71      @Test
72      public void verifySimpleNotValidWithSystemProperty() throws Exception {
73          File basedir = resources.getBasedir("simple");
74          Files.write(basedir.toPath().resolve("simple-java/src/main/resources/META-INF/beans.xml"),
75                  ("<beans xmlns=\"http://xmlns.jcp.org/xml/ns/javaee\" bean-discovery-mode=\"all\">\n"
76                          + "<scan><exclude name=\"test.FooImpl\"><if-system-property name=\"additional.test.property\"/></exclude></scan>\n"
77                          + "</beans>").getBytes(StandardCharsets.UTF_8));
78  
79          MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "verify");
80  
81          result.assertLogText("Unsatisfied dependencies for type Foo with qualifiers @Default")
82                  .assertLogText("private test.TestBean.foo");
83      }

95      public void verifyEjbsNotValid() throws Exception {
96          File basedir = resources.getBasedir("ejbs");
97          Files.write(basedir.toPath().resolve("simple-java/src/main/resources/META-INF/beans.xml"), new byte[0]);
98  
99          MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "verify");

106     @Test
107     public void verifyEjbsValidSecondEjbLocalBean() throws Exception {
108         File basedir = resources.getBasedir("ejbs");
109         Path ejbDescriptor = basedir.toPath().resolve("simple-ejb/src/main/resources/META-INF/ejb-jar.xml");
110         Files.createDirectories(ejbDescriptor.getParent());
111         Files.write(ejbDescriptor, Arrays.asList(
112                 "<ejb-jar version=\"3.1\" xmlns=\"http://java.sun.com/xml/ns/javaee\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xsi:schemaLocation=\"http://java.sun.com/xml/ns/javaee http://java.sun.com/xml/ns/javaee/ejb-jar_3_1.xsd\">",
113                 "<enterprise-beans>", //
114                 "<session>", //
115                 "<ejb-name>OtherBean</ejb-name>", //
116                 "<ejb-class>test.FooBean</ejb-class>", //
117                 "<local-bean/>", //
118                 "</session>", //
119                 "</enterprise-beans>", //
120                 "</ejb-jar>"), StandardCharsets.UTF_8);
121 
122         MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "verify");
123 
124         result.assertErrorFreeLog().assertLogText("local interfaces are [Foo]")
125                 .assertLogText("local interfaces are [FooBean]");
126     }

159     @Test
160     public void verifyWarsNotValid() throws Exception {
161         File basedir = resources.getBasedir("wars");
162         Path javaBean = basedir.toPath().resolve("simple-war2/src/main/java/test/FooBean.java");
163         List<String> lines = Files.readAllLines(javaBean, StandardCharsets.UTF_8);
164         lines.add(7, "@javax.ejb.LocalBean");
165         lines.add(7, "@javax.ejb.Local(FooBean.class)");
166         Files.write(javaBean, lines, StandardCharsets.UTF_8);
167 
168         MavenExecutionResult result = mavenRuntime.forProject(basedir).execute("clean", "verify");
169 
170         result.assertLogText("Unsatisfied dependencies for type Foo with qualifiers @Default")
171                 .assertLogText("protected test.TestImpl(Foo)").assertLogText("SKIPPED");
172     }
