23  public abstract class AbstractDeployment extends WeldDeployment {
24  
25      private final InjectionServices injectionServices = new MockInjectionServices();
26  
27      public AbstractDeployment(ResourceLoader resourceLoader, Bootstrap bootstrap,
28              Set<WeldBeanDeploymentArchive> beanDeploymentArchives, Iterable<Metadata<Extension>> extensions) {
29          super(resourceLoader, bootstrap, beanDeploymentArchives, extensions);
30          beanDeploymentArchives
31                  .forEach(archive -> archive.getServices().add(InjectionServices.class, injectionServices));
32          registerMockServices(getServices());
33      }
34  
35      private void registerMockServices(ServiceRegistry services) {
36          services.add(SecurityServices.class, new MockSecurityServices());
37          services.add(TransactionServices.class, new MockTransactionServices());
38          services.add(InjectionServices.class, injectionServices);
39          services.add(EjbServices.class, new MockEjbServices());
40      }

42  }
