20  import java.io.File;
21  import java.lang.reflect.InvocationTargetException;
22  import java.net.MalformedURLException;
23  import java.util.Set;
24  
25  import javax.enterprise.inject.spi.Extension;
26  import javax.validation.Validator;
27  import javax.validation.ValidatorFactory;
28  
29  import org.apache.maven.plugin.MojoExecutionException;
30  import org.jboss.weld.bootstrap.api.CDI11Bootstrap;
31  import org.jboss.weld.bootstrap.spi.Deployment;
32  import org.jboss.weld.bootstrap.spi.Metadata;
33  import org.jboss.weld.bootstrap.spi.helpers.MetadataImpl;
34  
35  import com.github.jonasrutishauser.cdi.maven.plugin.weld.bootstrap.PredefinedBeansExtension;
36  
37  public abstract class ArchiveUtil {
38  
39      public abstract void init(File workDirectory, File archive, ClassLoader parentClassLoader)
40              throws MalformedURLException;
41  
42      public abstract Deployment createDeployment(CDI11Bootstrap bootstrap) throws MojoExecutionException;
43  
44      public abstract ClassLoader getClassLoader();
45  
46      protected void addDefaultExtensions(Set<Metadata<Extension>> extensions) throws MojoExecutionException {
47          extensions.add(MetadataImpl.from(new PredefinedBeansExtension(Validator.class, ValidatorFactory.class)));
48          try {
49              extensions.add(
50                      MetadataImpl.from(getClassLoader().loadClass("io.smallrye.jwt.auth.cdi.SmallRyeJWTAuthCDIExtension")
51                              .asSubclass(Extension.class).getDeclaredConstructor().newInstance()));
52          } catch (ClassNotFoundException e) {
53              // no microprofile
54          } catch (InstantiationException | IllegalAccessException | InvocationTargetException
55                  | NoSuchMethodException e) {
56              throw new MojoExecutionException("Failed to create JWT extension", e);
57          }
58      }
59  
60  }
