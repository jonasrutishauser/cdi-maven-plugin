57      private final Set<File> ejbs = new HashSet<>();
58      private final Set<File> libraries = new HashSet<>();
59      private final Set<WarUtil> wars = new HashSet<>();

61      private ClassLoader earClassloader;

63      public EarUtil(ArchiverManager archiverManager) {
64          this.archiverManager = archiverManager;
65      }
66  
67      public Deployment createDeployment(CDI11Bootstrap bootstrap) {
68          ResourceLoader resourceLoader = new ClassLoaderResourceLoader(earClassloader);
69          Set<Metadata<Extension>> extensions = getExtensions(bootstrap);
70          TypeDiscoveryConfiguration typeDiscoveryConfiguration = bootstrap.startExtensions(extensions);
71          return new EarDeployment(resourceLoader, bootstrap, getBeanDeploymentArchives(resourceLoader, bootstrap,
72                  typeDiscoveryConfiguration.getKnownBeanDefiningAnnotations()), extensions);
73      }
74  
75      private Set<WeldBeanDeploymentArchive> getBeanDeploymentArchives(ResourceLoader resourceLoader,
76              CDI11Bootstrap bootstrap, Set<Class<? extends Annotation>> beanDefiningAnnotations) {
77          return Stream
78                  .of(getEjbArchives(resourceLoader, bootstrap, beanDefiningAnnotations),
79                          getLibraryArchives(resourceLoader, bootstrap, beanDefiningAnnotations),
80                          getWarArchives(bootstrap, beanDefiningAnnotations))
81                  .flatMap(Set::stream).collect(Collectors.toCollection(HashSet::new));
82      }
83  
84      private Set<WeldBeanDeploymentArchive> getWarArchives(CDI11Bootstrap bootstrap,
85              Set<Class<? extends Annotation>> beanDefiningAnnotations) {
86          return wars.stream().map(util -> util.createArchives(bootstrap, beanDefiningAnnotations)).flatMap(Set::stream)
87                  .collect(Collectors.toSet());
88      }
89  
90      private Set<WeldBeanDeploymentArchive> getLibraryArchives(ResourceLoader resourceLoader, CDI11Bootstrap bootstrap,
91              Set<Class<? extends Annotation>> beanDefiningAnnotations) {
92          DiscoveryStrategy strategy = DiscoveryStrategyFactory.create(resourceLoader, bootstrap, beanDefiningAnnotations,
93                  false);
94          strategy.setScanner(new JarsBeanArchiveScanner(bootstrap, libraries));
95          return strategy.performDiscovery().stream().peek(archive -> archive.getServices().add(EEModuleDescriptor.class,
96                  new EEModuleDescriptorImpl(archive.getId(), ModuleType.EAR))).collect(Collectors.toSet());
97      }
98  
99      private Set<WeldBeanDeploymentArchive> getEjbArchives(ResourceLoader resourceLoader, CDI11Bootstrap bootstrap,
100             Set<Class<? extends Annotation>> beanDefiningAnnotations) {
101         Set<Class<? extends Annotation>> annotations = new HashSet<>(beanDefiningAnnotations);
102         ejbUtil.addBeanDefiningAnnotations(annotations);
103         DiscoveryStrategy strategy = DiscoveryStrategyFactory.create(resourceLoader, bootstrap, annotations, false);
104         strategy.setScanner(new JarsBeanArchiveScanner(bootstrap, ejbs));
105         return strategy.performDiscovery().stream().map(ejbUtil::addEjbDescriptors).collect(Collectors.toSet());
106     }
107 
108     private Set<Metadata<Extension>> getExtensions(CDI11Bootstrap bootstrap) {
109         return Stream
110                 .of(Collections.singleton(earClassloader),
111                         wars.stream().map(WarUtil::getClassLoader).collect(Collectors.toSet()))
112                 .flatMap(Collection::stream).flatMap(classloader -> StreamSupport
113                         .stream(bootstrap.loadExtensions(classloader).spliterator(), false))
114                 .collect(Collectors.toSet());
115     }

117     @Override
118     public void init(File workDirectory, File earFile) throws MalformedURLException {
119         UnArchiver unArchiver;
120         try {
121             unArchiver = archiverManager.getUnArchiver(earFile);
122         } catch (NoSuchArchiverException e) {
123             throw new IllegalStateException(e);
124         }
125         File extractedDir = new File(workDirectory, earFile.getName());
126         extractedDir.mkdirs();
127         unArchiver.setDestDirectory(extractedDir);
128         unArchiver.setSourceFile(earFile);
129         unArchiver.extract();
130         File libDir = new File(extractedDir, "lib");
131         try {
132             DocumentBuilder documentBuilder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
133             Document doc = documentBuilder.parse(new File(extractedDir, "META-INF/application.xml"));
134             XPath xPath = XPathFactory.newInstance().newXPath();
135             NodeList modules = (NodeList) xPath.evaluate("//module/*", doc, XPathConstants.NODESET);
136             for (int i = 0; i < modules.getLength(); i++) {
137                 Element module = (Element) modules.item(i);
138                 if ("ejb".equals(module.getNodeName())) {
139                     ejbs.add(new File(extractedDir, module.getTextContent()));
140                 } else if ("web".equals(module.getNodeName())) {
141                     File warFile = new File(extractedDir,
142                             (String) xPath.evaluate("web-uri/text()", module, XPathConstants.STRING));
143                     wars.add(WarUtil.create(archiverManager, workDirectory, warFile));
144                 } else {
145                     libraries.add(new File(extractedDir, module.getTextContent()));
146                 }
147             }
148             Element libraryDirectory = (Element) xPath.evaluate("//library-directory", doc, XPathConstants.NODE);
149             if (libraryDirectory != null) {
150                 libDir = new File(extractedDir, libraryDirectory.getTextContent());
151             }
152         } catch (ParserConfigurationException | SAXException | XPathException | IOException e) {
153             LoggerFactory.getLogger(getClass()).warn("failed to read deployment descriptor of ear", e);
154         }
155         File[] jars = libDir.listFiles(file -> file.getName().endsWith(".jar"));
156         if (jars != null) {
157             for (File jar : jars) {
158                 libraries.add(jar);
159             }
160         }
161         List<URL> urls = new ArrayList<>(ejbs.size() + libraries.size());
162         for (File ejb : ejbs) {
163             urls.add(ejb.toURI().toURL());
164         }
165         for (File library : libraries) {
166             urls.add(library.toURI().toURL());
167         }
168         earClassloader = new URLClassLoader(urls.toArray(new URL[urls.size()]), getClass().getClassLoader());
169     }
