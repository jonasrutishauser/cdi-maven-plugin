20  import java.io.File;
21  import java.net.URL;
22  import java.util.Collection;
23  import java.util.List;
24  import java.util.Optional;
25  import java.util.stream.Collectors;
26  
27  import org.jboss.weld.bootstrap.api.Bootstrap;
28  import org.jboss.weld.bootstrap.spi.BeansXml;
29  import org.jboss.weld.environment.deployment.AbstractWeldDeployment;
30  import org.jboss.weld.environment.deployment.discovery.AbstractBeanArchiveScanner;
31  
32  public class JarsBeanArchiveScanner extends AbstractBeanArchiveScanner {
33      private final Collection<File> jars;
34  
35      public JarsBeanArchiveScanner(Bootstrap bootstrap, Collection<File> jars) {
36          super(bootstrap);
37          this.jars = jars;
38      }
39  
40      @Override
41      public List<ScanResult> scan() {
42          return jars.stream().map(this::getScanResult).filter(Optional::isPresent).map(Optional::get)
43                  .collect(Collectors.toList());
44      }
45  
46      private Optional<ScanResult> getScanResult(File jar) {
47          String beanArchiveRef = jar.getPath();
48          try {
49              BeansXml beansXml = parseBeansXml(new URL("jar:" + jar.toURI() + "!/" + AbstractWeldDeployment.BEANS_XML));
50              return accept(beansXml) ? Optional.of(new ScanResult(beansXml, beanArchiveRef))
51                      : Optional.empty();
52          } catch (Exception e) {
53              return Optional.of(new ScanResult(null, beanArchiveRef));
54          }
55      }
56  }
