3   import java.io.File;
4   import java.net.URL;
5   import java.util.Collection;
6   import java.util.List;
7   import java.util.Optional;
8   import java.util.stream.Collectors;
9   
10  import org.jboss.weld.bootstrap.api.Bootstrap;
11  import org.jboss.weld.bootstrap.spi.BeansXml;
12  import org.jboss.weld.environment.deployment.AbstractWeldDeployment;
13  import org.jboss.weld.environment.deployment.discovery.AbstractBeanArchiveScanner;
14  
15  public class JarsBeanArchiveScanner extends AbstractBeanArchiveScanner {
16      private final Collection<File> jars;
17  
18      public JarsBeanArchiveScanner(Bootstrap bootstrap, Collection<File> jars) {
19          super(bootstrap);
20          this.jars = jars;
21      }
22  
23      @Override
24      public List<ScanResult> scan() {
25          return jars.stream().map(this::getScanResult).filter(Optional::isPresent).map(Optional::get)
26                  .collect(Collectors.toList());
27      }
28  
29      private Optional<ScanResult> getScanResult(File jar) {
30          String beanArchiveRef = jar.getPath();
31          try {
32              BeansXml beansXml = parseBeansXml(new URL("jar:" + jar.toURI() + "!/" + AbstractWeldDeployment.BEANS_XML));
33              return accept(beansXml) ? Optional.of(new ScanResult(beansXml, beanArchiveRef))
34                      : Optional.empty();
35          } catch (Exception e) {
36              return Optional.of(new ScanResult(null, beanArchiveRef));
37          }
38      }
39  }
