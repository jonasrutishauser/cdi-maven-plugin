9   import java.util.List;
10  import java.util.stream.Collectors;
11  
12  import org.jboss.weld.bootstrap.api.Bootstrap;
13  import org.jboss.weld.bootstrap.spi.BeansXml;
14  import org.jboss.weld.environment.deployment.AbstractWeldDeployment;
15  
16  public final class WarBeanArchiveScanner extends JarsBeanArchiveScanner {
17  
18      public static final String CLASSES = "WEB-INF/classes/";
19      private static final String BEANS_XML = "WEB-INF/beans.xml";
20  
21      private final File war;
22  
23      public WarBeanArchiveScanner(Bootstrap bootstrap, File war, URL[] classLoaderUrls) {
24          super(bootstrap, Arrays.stream(classLoaderUrls).filter(WarBeanArchiveScanner::isWarLibrary).map(URL::toString)
25                  .map(URI::create).map(URI::getSchemeSpecificPart).map(File::new).collect(Collectors.toSet()));
26          this.war = war;
27      }
28  
29      private static boolean isWarLibrary(URL url) {
30          return !url.getPath().endsWith(CLASSES);
31      }
32  
33      @Override
34      public List<ScanResult> scan() {
35          File beansXmlFile = new File(war, BEANS_XML);
36          if (!beansXmlFile.exists()) {
37              beansXmlFile = new File(war, CLASSES + AbstractWeldDeployment.BEANS_XML);
38          }
39          BeansXml beansXml;
40          try {
41              beansXml = beansXmlFile.exists() ? parseBeansXml(beansXmlFile.toURI().toURL()) : null;
42          } catch (MalformedURLException e) {
43              throw new IllegalStateException(e);
44          }
45          List<ScanResult> result = new ArrayList<>();
46          result.add(new ScanResult(beansXml, new File(war, CLASSES).toString() + "/"));
47          result.addAll(super.scan());
48          return result;
49      }
50  
51  }
