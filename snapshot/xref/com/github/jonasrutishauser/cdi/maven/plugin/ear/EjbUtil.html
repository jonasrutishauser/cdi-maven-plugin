46      public void addBeanDefiningAnnotations(Set<Class<? extends Annotation>> annotations) {
47          annotations.add(Stateless.class);
48          annotations.add(Stateful.class);
49          annotations.add(Singleton.class);

53      public WeldBeanDeploymentArchive addEjbDescriptors(WeldBeanDeploymentArchive archive) {
54          archive.getServices().add(EEModuleDescriptor.class,
55                  new EEModuleDescriptorImpl(archive.getId(), ModuleType.EJB_JAR));
56          Predicate<Class<?>> isEjb = type -> type.isAnnotationPresent(Stateless.class)
57                  || type.isAnnotationPresent(Stateful.class) //
58                  || type.isAnnotationPresent(Singleton.class) //
59                  || type.isAnnotationPresent(MessageDriven.class);
60          ResourceLoader resourceLoader = archive.getServices().get(ResourceLoader.class);
61          Map<String, EjbDescriptor<?>> descriptors = archive.getBeanClasses().stream().map(resourceLoader::classForName)
62                  .filter(isEjb).collect(Collectors.toMap(Class::getSimpleName, SimpleEjbDescriptor::new, (u, v) -> {
63                      throw new IllegalStateException(String.format("Duplicate key %s", u));
64                  }, HashMap::new));
65          String archivePath = archive.getId();
66          if (archivePath.endsWith(WarBeanArchiveScanner.CLASSES)) {
67              addEjbsFromDescriptor(descriptors, resourceLoader,
68                      new File(new File(archivePath).getParentFile(), "ejb-jar.xml").toURI().toString());
69          } else {
70              addEjbsFromDescriptor(descriptors, resourceLoader,
71                      "jar:" + new File(archivePath).toURI() + "!/META-INF/ejb-jar.xml");
72          }
73          WeldBeanDeploymentArchive weldBeanDeploymentArchive = new WeldBeanDeploymentArchive(archive.getId(), archive.getBeanClasses(), archive.getBeansXml()) {
74              @Override
75              public Collection<EjbDescriptor<?>> getEjbs() {
76                  return descriptors.values();
77              }
78          };
79          weldBeanDeploymentArchive.getServices().addAll(archive.getServices().entrySet());
80          return weldBeanDeploymentArchive;
81      }

83      private void addEjbsFromDescriptor(Map<String, EjbDescriptor<?>> ejbs, ResourceLoader resourceLoader,
84              String descriptorUrl) {
85          try {
86              DocumentBuilder documentBuilder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
87              Document doc = documentBuilder.parse(descriptorUrl);
88              XPath xPath = XPathFactory.newInstance().newXPath();
89              NodeList beans = (NodeList) xPath.evaluate("//enterprise-beans/*", doc, XPathConstants.NODESET);
90              for (int i = 0; i < beans.getLength(); i++) {
91                  Element bean = (Element) beans.item(i);
92                  String name = getValue(bean, "ejb-name").get();
93                  SimpleEjbDescriptor<?> descriptor = (SimpleEjbDescriptor<?>) ejbs.computeIfAbsent(name,
94                          beanName -> new SimpleEjbDescriptor<>(
95                                  resourceLoader.classForName(getValue(bean, "ejb-class").get()), beanName));
96                  boolean hasInterface = hasInterface(bean);
97                  if (hasInterface) {
98                      descriptor.getLocalBusinessInterfaces().clear();
99                      if (getValue(bean, "local-bean").isPresent()) {
100                         descriptor.addLocalInterface(descriptor.getBeanClass());
101                     }
102                     NodeList interfaces = bean.getElementsByTagName("business-local");
103                     for (int j = 0; j < interfaces.getLength(); j++) {
104                         descriptor.addLocalInterface(resourceLoader.classForName(interfaces.item(j).getTextContent()));
105                     }
106                 }
107             }
108         } catch (ParserConfigurationException | SAXException | XPathException e) {
109             LoggerFactory.getLogger(getClass()).warn("failed to read deployment descriptor in " + descriptorUrl, e);
110         } catch (IOException e) {
111             // ignore
112         }
113     }
114 
115     private boolean hasInterface(Element bean) {
116         Set<String> interfaces = new HashSet<>(Arrays.asList("home", "remote", "local-home", "local", "business-local",
117                 "business-remote", "local-bean"));
118         for (int i = 0; i < bean.getChildNodes().getLength(); i++) {
119             Node node = bean.getChildNodes().item(i);
120             if (node instanceof Element && interfaces.contains(node.getNodeName())) {
121                 return true;
122             }
123         }
124         return false;
125     }
126 
127     private Optional<String> getValue(Element element, String elementName) {
128         NodeList elements = element.getElementsByTagName(elementName);
129         if (elements.getLength() == 1) {
130             return Optional.of(elements.item(0).getTextContent());
131         }
132         return Optional.empty();
133     }
