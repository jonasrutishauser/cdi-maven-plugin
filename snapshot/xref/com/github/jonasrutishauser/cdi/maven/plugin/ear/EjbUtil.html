63      public void addBeanDefiningAnnotations(Set<Class<? extends Annotation>> annotations) {
64          annotations.add(Stateless.class);
65          annotations.add(Stateful.class);
66          annotations.add(Singleton.class);
67          annotations.add(MessageDriven.class);
68      }

70      public WeldBeanDeploymentArchive addEjbDescriptors(WeldBeanDeploymentArchive archive) {
71          archive.getServices().add(EEModuleDescriptor.class,
72                  new EEModuleDescriptorImpl(archive.getId(), ModuleType.EJB_JAR));
73          Predicate<Class<?>> isEjb = type -> type.isAnnotationPresent(Stateless.class)
74                  || type.isAnnotationPresent(Stateful.class) //
75                  || type.isAnnotationPresent(Singleton.class) //
76                  || type.isAnnotationPresent(MessageDriven.class);
77          ResourceLoader resourceLoader = archive.getServices().get(ResourceLoader.class);
78          Map<String, EjbDescriptor<?>> descriptors = archive.getBeanClasses().stream().map(resourceLoader::classForName)
79                  .filter(isEjb).collect(Collectors.toMap(Class::getSimpleName, SimpleEjbDescriptor::new, (u, v) -> {
80                      throw new IllegalStateException(String.format("Duplicate key %s", u));
81                  }, HashMap::new));
82          String archivePath = archive.getId();
83          if (archivePath.endsWith(WarBeanArchiveScanner.CLASSES)) {
84              addEjbsFromDescriptor(descriptors, resourceLoader,
85                      new File(new File(archivePath).getParentFile(), "ejb-jar.xml").toURI().toString());
86          } else {
87              addEjbsFromDescriptor(descriptors, resourceLoader,
88                      "jar:" + new File(archivePath).toURI() + "!/META-INF/ejb-jar.xml");
89          }
90          WeldBeanDeploymentArchive weldBeanDeploymentArchive = new WeldBeanDeploymentArchive(archive.getId(),
91                  archive.getBeanClasses(), archive.getBeansXml()) {
92              @Override
93              public Collection<EjbDescriptor<?>> getEjbs() {
94                  return descriptors.values();
95              }
96          };
97          weldBeanDeploymentArchive.getServices().addAll(archive.getServices().entrySet());
98          return weldBeanDeploymentArchive;
99      }

101     private void addEjbsFromDescriptor(Map<String, EjbDescriptor<?>> ejbs, ResourceLoader resourceLoader,
102             String descriptorUrl) {
103         try {
104             DocumentBuilder documentBuilder = DocumentBuilderFactory.newInstance().newDocumentBuilder();
105             Document doc = documentBuilder.parse(descriptorUrl);
106             XPath xPath = XPathFactory.newInstance().newXPath();
107             NodeList beans = (NodeList) xPath.evaluate("//enterprise-beans/*", doc, XPathConstants.NODESET);
108             for (int i = 0; i < beans.getLength(); i++) {
109                 Element bean = (Element) beans.item(i);
110                 String name = getValue(bean, "ejb-name").orElseThrow(
111                         () -> new IllegalArgumentException("required 'ejb-name' not found in deployment descriptor"));
112                 SimpleEjbDescriptor<?> descriptor = (SimpleEjbDescriptor<?>) ejbs.computeIfAbsent(name,
113                         beanName -> new SimpleEjbDescriptor<>(
114                                 resourceLoader.classForName(
115                                         getValue(bean, "ejb-class").orElseThrow(() -> new IllegalArgumentException(
116                                                 "required 'ejb-class' not found in deployment descriptor"))),
117                                 beanName));
118                 boolean hasInterface = hasInterface(bean);
119                 if (hasInterface) {
120                     descriptor.getLocalBusinessInterfaces().clear();
121                     if (getValue(bean, "local-bean").isPresent()) {
122                         descriptor.addLocalInterface(descriptor.getBeanClass());
123                     }
124                     NodeList interfaces = bean.getElementsByTagName("business-local");
125                     for (int j = 0; j < interfaces.getLength(); j++) {
126                         descriptor.addLocalInterface(resourceLoader.classForName(interfaces.item(j).getTextContent()));
127                     }
128                 }
129             }
130         } catch (ParserConfigurationException | SAXException | XPathException e) {
131             LoggerFactory.getLogger(getClass()).warn("failed to read deployment descriptor in " + descriptorUrl, e);
132         } catch (IOException e) {
133             // ignore
134         }
135     }
136 
137     private boolean hasInterface(Element bean) {
138         Set<String> interfaces = new HashSet<>(Arrays.asList("home", "remote", "local-home", "local", "business-local",
139                 "business-remote", "local-bean"));
140         for (int i = 0; i < bean.getChildNodes().getLength(); i++) {
141             Node node = bean.getChildNodes().item(i);
142             if (node instanceof Element && interfaces.contains(node.getNodeName())) {
143                 return true;
144             }
145         }
146         return false;
147     }
148 
149     private Optional<String> getValue(Element element, String elementName) {
150         NodeList elements = element.getElementsByTagName(elementName);
151         if (elements.getLength() == 1) {
152             return Optional.of(elements.item(0).getTextContent());
153         }
154         return Optional.empty();
155     }
